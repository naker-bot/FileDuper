cmake_minimum_required(VERSION 3.16)
project(FileDuper_Professional)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find Qt6
find_package(Qt6 REQUIRED COMPONENTS Core Widgets)

# Enable automatic MOC, UIC and RCC
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

# Create executable
add_executable(FileDuper_Professional
    src_minimal/mainwindow_professional.cpp
)

# Link Qt6 libraries
target_link_libraries(FileDuper_Professional
    Qt6::Core
    Qt6::Widgets
)

# Installation
install(TARGETS FileDuper_Professional
    RUNTIME DESTINATION bin
)

# Set output name
set_target_properties(FileDuper_Professional PROPERTIES
    OUTPUT_NAME "FileDuper_Professional"
)

message(STATUS "âœ… FileDuper Professional configured successfully")
