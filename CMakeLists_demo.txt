cmake_minimum_required(VERSION 3.16)
project(DirectoryTreeDemo)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Core Widgets)

qt6_standard_project_setup()

qt6_add_executable(directory_tree_demo
    main_demo.cpp
    directorytreewidget.cpp
    directorytreewidget.h
)

qt6_add_resources(directory_tree_demo "icons"
    PREFIX "/"
    BASE "assets"
    FILES
        assets/icons/folder.svg
)

target_link_libraries(directory_tree_demo Qt6::Core Qt6::Widgets)
